/**
 * Interface used by the LifeIO class to access a Life processor
 *
 * @type {Class}
 */
(function() {
    /**
     * @interface
     * @constructor
     */
    function LifeIO() {}
    /**
     * Create a new Life game
     *
     * @param {number} width width of the Life game (X axis)
     * @param {number} height height of the Life game (Y axis)
     * @param {Function} cellChanges function to be called whenever there are cell changes to be drawn
     */
    LifeIO.prototype.create=function(width, height, cellChanges) {};

    /**
     * Restores the current state of a Life game
     *
     * @param id {string} identifying unique code of the Life game to be restored
     * @param sizeRecorder {Function} to be called with the Life game's dimension (parameter: {width:number,height:number})
     * @param cellChanges {Function} to be called whenever there are cell changes to be drawn
     */
    LifeIO.prototype.restore=function(id, sizeRecorder, cellChanges) {};

    /**
     * Loads the current state of the Life game.
     *
     * @param cellChanges {Function} to be called with the cell changes and any subsequent changes
     */
    LifeIO.prototype.load=function(cellChanges) {};

    /**
     * Stops the processing of this Life game
     */
    LifeIO.prototype.stop=function() {};

    /**
     * Resizes the Life game
     *
     * @param {number} width width of the Life game (X axis)
     * @param {number} height height of the Life game (Y axis)
     */
    LifeIO.prototype.resize=function(width, height) {};

    /**
     * Applies a set of cells (alive or dead) into the Life game
     *
     * @param {Array.Object} cells An Array of cell values  where v is true or false for alive or dead)
     */
    LifeIO.prototype.apply=function(cells) {}
})();
