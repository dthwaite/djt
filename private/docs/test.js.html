/**
 * Bouncer class constructor
 *
 * @param obj - The object that is dong the bouncing
 * @param params - parameters to affect the environment
 * @constructor
 * @class A Bouncer instance manages one object bouncing within its parent
 */
function Bouncer(obj,params) {
    /**
     * Stop or Starts animation
     */
    this.stopgo = function (command) {
        switch (command) {
            case "stop":
                params.paused = true;
                break;
            case "go":
                params.paused = false;
                break;
            default:
                params.paused = !params.paused;
                break;
        }
        if (params.paused) obj.stop();
        else animate();
    };
}
